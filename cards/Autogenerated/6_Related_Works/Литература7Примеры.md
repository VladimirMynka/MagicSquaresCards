$$
\begin{matrix}
17^2 & 5^2 \cdot 7^2 & 19^2 \\
17 \cdot 41 & 5^4 & 7 \cdot 79 \\
7 \cdot 127 & 5^2 & 31^2
\end{matrix}
$$

$$
\begin{matrix}
5^2 \cdot 7^2 & 17^2 & 19^2 \\
-239 & 5^4 & 1489 \\
7 \cdot 127 & 31^2 & 5^2
\end{matrix}
$$
