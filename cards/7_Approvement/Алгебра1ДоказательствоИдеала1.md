```
sm(E₀, 0, 0, 0, 0) * sm(E, x, y, v, w) = sm(
  3E₀E,
  3*0*w + 3*x*0 - 3*0*v + 3*y*0,
  -3*0*v + 3*x*0 + 3*0*w + 3*y*0,
    0*x -   0*y + 3*0*w + 3*0*v,
    -0*y +   x*0 + 3*0*v + 3*w*0,
) = sm(3E₀E, 0, 0, 0, 0)
